List Operations Demo

Java collections framework with ArrayList operations and sorting

import java.util.*;

public class SimpleListDemo {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        List list = new ArrayList<>();

        // Step 1: Accept initial list
        System.out.print("Enter number of elements: ");
        int n = sc.nextInt();
        sc.nextLine();
        for (int i = 0; i < n; i++) {
            System.out.print("Enter element " + (i + 1) + ": ");
            list.add(sc.nextLine());
        }

        int choice;
        // Step 2: Menu-driven methods
        do {
            System.out.println("\n--- List Menu ---");
            System.out.println("1. Add element");
            System.out.println("2. Get element by index");
            System.out.println("3. Update element (set)");
            System.out.println("4. Remove element");
            System.out.println("5. Check if element exists");
            System.out.println("6. Show full list & Exit");
            System.out.print("Enter choice: ");
            choice = sc.nextInt();
            sc.nextLine();

            switch (choice) {
                case 1: // add()
                    System.out.print("Enter element to add: ");
                    list.add(sc.nextLine());
                    break;

                case 2: // get()
                    System.out.print("Enter index: ");
                    int idx = sc.nextInt();
                    if (idx >= 0 && idx < list.size())
                        System.out.println("Element: " + list.get(idx));
                    else
                        System.out.println("Invalid index!");
                    break;

                case 3: // set()
                    System.out.print("Enter index: ");
                    int setIdx = sc.nextInt();
                    sc.nextLine();
                    if (setIdx >= 0 && setIdx < list.size()) {
                        System.out.print("Enter new value: ");
                        list.set(setIdx, sc.nextLine());
                    } else
                        System.out.println("Invalid index!");
                    break;

                case 4: // remove()
                    System.out.print("Enter element to remove: ");
                    list.remove(sc.nextLine());
                    break;

                case 5: // contains()
                    System.out.print("Enter element to check: ");
                    System.out.println("Exists? " + list.contains(sc.nextLine()));
                    break;

                case 6: // exit
                    System.out.println("Final List: " + list);
                    System.out.println("Exiting...");
                    break;
            }
        } while (choice != 6);

        sc.close();
    }
}
