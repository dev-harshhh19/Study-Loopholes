Method Overloading & Overriding Demo

Demonstrates method overloading and overriding with Calculator classes

// Parent class
class Calculator {
    // Method Overloading (same name, different parameters)
    int add(int a, int b) {
        return a + b;
    }

    int add(int a, int b, int c) {
        return a + b + c;
    }

    // Method to be overridden
    void show() {
        System.out.println("Calculator class");
    }
}

// Child class
class AdvancedCalculator extends Calculator {
    // Method Overriding (same name, same parameters, different class)
    @Override
    void show() {
        System.out.println("AdvancedCalculator class");
    }
}

public class MethodDemo {
    public static void main(String[] args) {
        Calculator c = new Calculator();
        System.out.println("Sum of 2 numbers: " + c.add(5, 10));       // Overloaded method
        System.out.println("Sum of 3 numbers: " + c.add(2, 3, 4));     // Overloaded method
        c.show();  // Parent version

        AdvancedCalculator ac = new AdvancedCalculator();
        ac.show(); // Overridden version
    }
}
