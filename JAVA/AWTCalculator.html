AWT Calculator

GUI Calculator using AWT components with event handling

import java.awt.*;
import java.awt.event.*;

public class CalculatorAWT extends Frame implements ActionListener {
    TextField display;
    String operator = "";
    double num1 = 0, num2 = 0, result = 0;

    CalculatorAWT() {
        // Display screen
        display = new TextField();
        display.setEditable(false);
        display.setFont(new Font("Arial", Font.PLAIN, 18)); // smaller font
        add(display, BorderLayout.NORTH);

        // Buttons
        String buttons[] = {
            "7", "8", "9", "/", 
            "4", "5", "6", "*", 
            "1", "2", "3", "-", 
            "0", "C", "=", "+"
        };

        Panel panel = new Panel();
        panel.setLayout(new GridLayout(4, 4, 3, 3)); // tighter spacing

        for (String b : buttons) {
            Button btn = new Button(b);
            btn.setFont(new Font("Arial", Font.PLAIN, 16)); // smaller buttons
            btn.addActionListener(this);
            panel.add(btn);
        }

        add(panel, BorderLayout.CENTER);

        // Frame settings (smaller window)
        setTitle("AWT Calculator");
        setSize(220, 280);  // reduced size
        setResizable(false);
        setVisible(true);

        // Close window
        addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent we) {
                System.exit(0);
            }
        });
    }

    public void actionPerformed(ActionEvent e) {
        String cmd = e.getActionCommand();

        if (cmd.matches("[0-9]")) {
            display.setText(display.getText() + cmd);
        }
        else if (cmd.matches("[+\\-*/]")) {
            num1 = Double.parseDouble(display.getText());
            operator = cmd;
            display.setText("");
        }
        else if (cmd.equals("=")) {
            num2 = Double.parseDouble(display.getText());
            switch (operator) {
                case "+": result = num1 + num2; break;
                case "-": result = num1 - num2; break;
                case "*": result = num1 * num2; break;
                case "/": result = (num2 != 0) ? num1 / num2 : 0; break;
            }
            display.setText(String.valueOf(result));
        }
        else if (cmd.equals("C")) {
            display.setText("");
            num1 = num2 = result = 0;
            operator = "";
        }
    }

    public static void main(String[] args) {
        new CalculatorAWT();
    }
}
