print('First come first serve scheduling')
n=int(input("enter number of process:"))
d=dict()
for i in range(n):
    key="p"+str(i+1)

    #first process

Shortest Job First (SJF)

class Process:
    def __init__(self,pid,arrival,burst):
        self.pid=pid              #process id
        self.arrival=arrival      #arrival time
        self.burst=burst          #burst time (time to execute)
        self.completion=0         #completion time (to be calculated)
        self.turnaround=0         #turnaround time (to be calculated)
        self.waiting=0            #waiting time (to be calculated)

#function to perform sjf (non-preemptive)
def sjf(processes):
    time=0             #start from time 0
    completed=0        #count of completed process
    grant=[]           #Grant chart by visualise execution order

    #step 1: sort the process by arrival time
    processes.sort(key=lambda p: p.arrival)

    #step 2: loop to schedule processes
    while completed < len(processes):
        #find the processes that arrived are not yet completed
        ready=[p for p in processes if p.arrival<=time and p.completion==0]

        if not ready:
            time+=1      #if no process is ready , move time forward
            continue

        #step 3: select the process with the shortest burst
        current=min(ready,key=lambda p:p.burst)
        grant.append(f"P{current.pid} ({time})")
        time+=current.burst

        #step 4: update process details after execution
        current.completion=time
        current.turnaround=current.completion-current.arrival
        current.waiting=current.turnaround-current.burst
        completed+=1   #mark process as completed

    #step 5: display the results in tabular format
    print("\nProcess Arrival Burst Completion Turnaround Waiting")
    total_turnaround=0
    total_waiting=0
    for p in processes:
        print(f"P{p.pid}          {p.arrival}          {p.burst}         {p.completion}         {p.turnaround}       {p.waiting}")
        total_turnaround+=p.turnaround
        total_waiting+=p.waiting

    #step 6: calculate and display average waiting and turnaround time
    avg_turnaround=total_turnaround/len(processes)
    avg_waiting=total_waiting/len(processes)
    print(f"\nAverage Turnaround Time : {avg_turnaround}")
    print(f"Average Waiting Time : {avg_waiting}")

    #step 7: display the grant chart (execution order)
    print("\nGrant Chart :")
    print(" -> ".join(grant))

#main function to take input and run the scheduling
if __name__=="__main__":
    n=int(input("Enter number of processes:"))
    processes=[]
    #input data for each process
    for i in range(n):
        arrival=int(input("Enter the arrival time of process "+str(i+1)+":"))
        burst=int(input("Enter the burst time of process "+str(i+1)+":"))
        processes.append(Process(i+1,arrival,burst))
    sjf(processes)

Shortest Remaining Time First (SRTF)

class Process:
    def __init__(self, pid, arrival, burst):
        self.pid = pid
        self.arrival = arrival
        self.burst = burst
        self.remaining = burst
        self.completion = 0
        self.turnaround = 0
        self.waiting = 0

def srtf(processes):
    time = 0
    completed = 0
    grant = []

    while completed < len(processes):
        # ready queue
        ready = [p for p in processes if p.arrival <= time and p.remaining > 0]

        if not ready:
            time += 1
            continue

        # select process with shortest remaining time
        current = min(ready, key=lambda p: p.remaining)
        grant.append(f"P{current.pid} ({time})")

        current.remaining -= 1
        time += 1

        if current.remaining == 0:
            completed += 1
            current.completion = time
            current.turnaround = current.completion - current.arrival
            current.waiting = current.turnaround - current.burst

    # Output table
    print("\nProcess Arrival Burst Completion Turnaround Waiting")
    for p in processes:
        print(f"P{p.pid:<8} {p.arrival:<7} {p.burst:<6} {p.completion:<10} {p.turnaround:<10} {p.waiting}")

    avg_turnaround = sum(p.turnaround for p in processes) / len(processes)
    avg_waiting = sum(p.waiting for p in processes) / len(processes)

    print("\nAvg Turnaround Time:", avg_turnaround)
    print("Avg Waiting Time:", avg_waiting)

    print("\nGrant Chart:", " -> ".join(grant))

if __name__ == "__main__":
    n = int(input("Enter number of processes: "))
    processes = []

    for i in range(n):
        arrival = int(input("Enter arrival time for process " + str(i+1) + ": "))
        burst = int(input("Enter burst time for process " + str(i+1) + ": "))
        processes.append(Process(i+1, arrival, burst))

    # Run SRTF scheduling
    srtf(processes)

Round Robin (RR)

from collections import deque

print("ROUND ROBIN SCHEDULING")

n = int(input("Enter number of processes: "))
d = {}

for i in range(n):
    key = "P" + str(i + 1)
    a = int(input(f"Enter arrival time for process {i+1}:"))
    b = int(input(f"Enter burst time for process {i+1}:"))
    d[key] = [a, b]  # [Arrival Time, Burst Time]

quantum = int(input("Enter Time Quantum: "))

# Sort processes by arrival time
d = dict(sorted(d.items(), key=lambda item: item[1][0]))

arrival = {}
burst = {}
remaining = {}
completed = {}
completion_time = {}
turnaround_time = {}
waiting_time = {}

for k, v in d.items():
    arrival[k] = v[0]
    burst[k] = v[1]
    remaining[k] = v[1]
    completed[k] = False

time = list(arrival.values())[0]  # start from earliest arrival
queue = deque()

# Enqueue first arriving processes
for k in d:
    if arrival[k] <= time:
        queue.append(k)

visited = set(queue)

while queue:
    process = queue.popleft()

    if remaining[process] > quantum:
        time += quantum
        remaining[process] -= quantum
    else:
        time += remaining[process]
        remaining[process] = 0
        completed[process] = True
        completion_time[process] = time

    # Add new processes that have arrived by now
    for k in d:
        if arrival[k] <= time and not completed[k] and k not in visited:
            queue.append(k)
            visited.add(k)

    # If current process is not done, add it back
    if not completed[process]:
        queue.append(process)

# Calculate Turnaround Time and Waiting Time
for k in d:
    turnaround_time[k] = completion_time[k] - arrival[k]
    waiting_time[k] = turnaround_time[k] - burst[k]

# Calculate averages
avg_tat = sum(turnaround_time.values()) / n
avg_wt = sum(waiting_time.values()) / n

# Print Output
print("\nProcess  Arrival  Burst  Completion  Turnaround  Waiting")
for k in d:
    print(f"{k:<8} {arrival[k]:<8} {burst[k]:<6} {completion_time[k]:<11} {turnaround_time[k]:<10} {waiting_time[k]}")

print("\nAvg Turnaround Time:", avg_tat)
print("Avg Waiting Time:", avg_wt)
