import time
import random 
from queue import Queue

class Process:
    def __init__(self, pid):
        self.pid = pid
        self.state = "New"

    def __str__(self):
            return f"Process {self.pid} is in {self.state} state"
        
class OS:
    def __init__(self):
        self.new = Queue()
        self.ready = Queue()
        self.waiting = Queue()
        self.running = None
        self.terminated = []

    def create_processs(self, pid):
        p = Process(pid)
        print(p)
        self.new.put(p)
    
    def admit_process(self):
        if not self.new.empty():
            p = self.new.get()
            p.state = "Ready"
            print(p)
            self.ready.put(p)
        
    def dispatched_process(self):
        if not self.ready.empty():
            p = self.ready.get()
            p.state = "Running"
            self.running = p
            print(p)
    
    def run_process(self):
        if self.running:
            action = random.choice(["terminate", "wait", "Continue"])
            if action == "terminate":
                self.running.state = "Terminated"
                print(self.running)
                self.terminated.append(self.running)
                self.running = None
            elif action == "wait":
                self.running.state = "Waiting"
                print(self.running)
                self.waiting.put(self.running)
                self.running = None
            else:
                print(f"{self.running} continue running")
    
    def wake_process(self):
        if not self.waiting.empty():
            p = self.waiting.get()
            p.state = "Ready"
            print(f"{p} is waking up and going to ready")
            self.ready.put(p)
    
    def step(self):
        self.admit_process()
        if not self.running:
            self.dispatched_process()
        self.run_process()
        self.wake_process()

def simulate():
        os = OS()
        pid = 1
        for _ in range(5):
            os.create_processs(pid)
            pid += 1
        
        for _ in range(15):
            print("\n---- OS Cycle ---")
            os.step()
            time.sleep(0.5)

simulate()
