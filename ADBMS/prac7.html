  • Program 1: Implicit and Explicit Cursor

Write an Implicit and explicit cursor to complete the task.

-- Step 1: Create table
CREATE TABLE employee (
    emp_id NUMBER,
    emp_name VARCHAR2(50),
    salary NUMBER
);

-- Step 2: Insert sample data
INSERT INTO employee VALUES (1, 'John', 30000);
INSERT INTO employee VALUES (2, 'Mary', 40000);
INSERT INTO employee VALUES (3, 'Alex', 50000);
COMMIT;

-- Step 3: Integrated PL/SQL Program
DECLARE
    -- Variables for Implicit Cursor
    v_total_salary NUMBER;
    v_emp_count NUMBER;

    -- Variables for Explicit Cursor
    v_id employee.emp_id%TYPE;
    v_name employee.emp_name%TYPE;

    -- Explicit Cursor
    CURSOR emp_cur IS SELECT emp_id, emp_name FROM employee;

    -- User-Defined Exception
    no_emp_found EXCEPTION;
BEGIN
    -- Implicit Cursor Example (Predefined Exception: ZERO_DIVIDE)
    SELECT SUM(salary), COUNT(*) INTO v_total_salary, v_emp_count FROM employee;

    IF v_emp_count = 0 THEN
        RAISE no_emp_found; -- User-defined exception
    END IF;

    DBMS_OUTPUT.PUT_LINE('Average Salary: ' || (v_total_salary / v_emp_count));

    -- Explicit Cursor Example
    OPEN emp_cur;
    LOOP
        FETCH emp_cur INTO v_id, v_name;
        EXIT WHEN emp_cur%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE('ID: ' || v_id || ', Name: ' || v_name);
    END LOOP;
    CLOSE emp_cur;

EXCEPTION
    WHEN ZERO_DIVIDE THEN
        DBMS_OUTPUT.PUT_LINE('Error: Division by zero occurred.');
    WHEN no_emp_found THEN
        DBMS_OUTPUT.PUT_LINE('Error: No employees found in table.');
END;
