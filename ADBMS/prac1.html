Program 1: IF...THEN...ELSE

Question:

Write a PL/SQL block to check whether a number is even or odd using IF...THEN...ELSE.

DECLARE
    num NUMBER := 7;
BEGIN
    IF MOD(num, 2) = 0 THEN
        DBMS_OUTPUT.PUT_LINE(num || ' is Even');
    ELSE
        DBMS_OUTPUT.PUT_LINE(num || ' is Odd');
    END IF;
END;
/

Program 2: IF...ELSIF...ELSE

Question:

Write a PL/SQL block to check if a number is positive, negative, or zero.

DECLARE
    num NUMBER := -15;
BEGIN
    IF num > 0 THEN
        DBMS_OUTPUT.PUT_LINE('Positive Number');
    ELSIF num < 0 THEN
        DBMS_OUTPUT.PUT_LINE('Negative Number');
    ELSE
        DBMS_OUTPUT.PUT_LINE('Zero');
    END IF;
END;
/

Program 3: Nested IF...ELSE

Question:

Write a PL/SQL block to check if a student has passed and also assign a grade.

DECLARE
    marks NUMBER := 85;
BEGIN
    IF marks >= 40 THEN
        IF marks >= 75 THEN
            DBMS_OUTPUT.PUT_LINE('Passed with Distinction');
        ELSIF marks >= 60 THEN
            DBMS_OUTPUT.PUT_LINE('Passed with First Class');
        ELSE
            DBMS_OUTPUT.PUT_LINE('Passed');
        END IF;
    ELSE
        DBMS_OUTPUT.PUT_LINE('Failed');
    END IF;
END;
/

Program 4: Simple CASE Statement

Question:

Write a PL/SQL block to display day name based on a number (1 to 3).

DECLARE
    day_num NUMBER := 2;
BEGIN
    CASE day_num
        WHEN 1 THEN DBMS_OUTPUT.PUT_LINE('Monday');
        WHEN 2 THEN DBMS_OUTPUT.PUT_LINE('Tuesday');
        WHEN 3 THEN DBMS_OUTPUT.PUT_LINE('Wednesday');
        ELSE DBMS_OUTPUT.PUT_LINE('Invalid Day');
    END CASE;
END;
/

Program 5: Searched CASE Statement

Question:

Write a PL/SQL block to assign grade based on marks using searched CASE.

DECLARE
    marks NUMBER := 72;
BEGIN
    CASE
        WHEN marks >= 75 THEN DBMS_OUTPUT.PUT_LINE('Distinction');
        WHEN marks >= 60 THEN DBMS_OUTPUT.PUT_LINE('First Class');
        WHEN marks >= 40 THEN DBMS_OUTPUT.PUT_LINE('Pass');
        ELSE DBMS_OUTPUT.PUT_LINE('Fail');
    END CASE;
END;
/

Program 6: Nested CASE Statement

Question:

Write a PL/SQL block to assign grade and print result status using nested CASE.

DECLARE
    marks NUMBER := 88;
    result VARCHAR2(20);
BEGIN
    result := CASE
        WHEN marks >= 40 THEN
            CASE
                WHEN marks >= 75 THEN 'Distinction'
                WHEN marks >= 60 THEN 'First Class'
                ELSE 'Pass'
            END;
        ELSE
            'Fail'
    END;
    DBMS_OUTPUT.PUT_LINE('Result: ' || result);
END;
