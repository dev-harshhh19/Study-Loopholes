Program 1: Sequential Statements – Simple Arithmetic

DECLARE
    a NUMBER := 10;
    b NUMBER := 5;
    sum NUMBER;
    diff NUMBER;
BEGIN
    sum := a + b;
    diff := a - b;
    DBMS_OUTPUT.PUT_LINE('Sum: ' || sum);
    DBMS_OUTPUT.PUT_LINE('Difference: ' || diff);
END;

Program 2: Sequential Statements – Area and Perimeter of Rectangle

DECLARE
    length NUMBER := 8;
    width NUMBER := 5;
    area NUMBER;
    perimeter NUMBER;
BEGIN
    area := length * width;
    perimeter := 2 * (length + width);
    DBMS_OUTPUT.PUT_LINE('Area: ' || area);
    DBMS_OUTPUT.PUT_LINE('Perimeter: ' || perimeter);
END;
/

Program 3: Unconstrained LOOP – Print 1 to 5

DECLARE
    i NUMBER := 1;
BEGIN
    LOOP
        EXIT WHEN i > 5;
        DBMS_OUTPUT.PUT_LINE(i);
        i := i + 1;
    END LOOP;
END;
/

Program 4: Unconstrained LOOP – Sum of Digits of a Number

DECLARE
    num NUMBER := 1234;
    digit NUMBER;
    sum_digits NUMBER := 0;
BEGIN
    LOOP
        EXIT WHEN num = 0;
        digit := MOD(num, 10);
        sum_digits := sum_digits + digit;
        num := TRUNC(num / 10);
    END LOOP;
    DBMS_OUTPUT.PUT_LINE('Sum of digits: ' || sum_digits);
END;
/

Program 5: Unconstrained LOOP – Find Smallest Divisor (Except 1)

DECLARE
    num NUMBER := 91;
    div NUMBER := 2;
BEGIN
    LOOP
        EXIT WHEN MOD(num, div) = 0;
        div := div + 1;
    END LOOP;
    DBMS_OUTPUT.PUT_LINE('Smallest Divisor of ' || num || ' is: ' || div);
END;
/

Program 6: Sequential Statements – Swap Two Numbers Using a Third Variable

DECLARE
    a NUMBER := 10;
    b NUMBER := 20;
    temp NUMBER;
BEGIN
    temp := a;
    a := b;
    b := temp;
    DBMS_OUTPUT.PUT_LINE('After Swapping: a = ' || a || ', b = ' || b);
END;
/

Program 7: Unconstrained LOOP – Print First 5 Odd Numbers

DECLARE
    i NUMBER := 1;
    count NUMBER := 0;
BEGIN
    LOOP
        EXIT WHEN count = 5;
        IF MOD(i, 2) = 1 THEN
            DBMS_OUTPUT.PUT_LINE(i);
            count := count + 1;
        END IF;
        i := i + 1;
    END LOOP;
END;
/

Program 8: Unconstrained LOOP – Count How Many Times Division by 2 is Possible

DECLARE
    num NUMBER := 64;
    count NUMBER := 0;
BEGIN
    LOOP
        EXIT WHEN MOD(num, 2) != 0;
        num := num / 2;
        count := count + 1;
    END LOOP;
    DBMS_OUTPUT.PUT_LINE('Divided by 2, ' || count || ' times until it became odd.');
END;
