Writing Functions in PL/SQL Block

-- Enable DBMS output
SET SERVEROUTPUT ON;

-- Create the employees table
CREATE TABLE employees (
    emp_id NUMBER,
    emp_name VARCHAR2(50),
    dept_id NUMBER
);

-- Insert sample data
BEGIN
    INSERT INTO employees VALUES (1, 'John Doe', 10);
    INSERT INTO employees VALUES (2, 'Jane Smith', 20);
    INSERT INTO employees VALUES (3, 'Alice Johnson', 10);
    COMMIT;
END;
/

-- Recursive factorial function
CREATE OR REPLACE FUNCTION factorial(n IN NUMBER) RETURN NUMBER IS
BEGIN
    IF n = 0 THEN
        RETURN 1;
    ELSE
        RETURN n * factorial(n - 1);
    END IF;
END;
/

-- Count employees in a department
CREATE OR REPLACE FUNCTION count_employees(dept NUMBER) RETURN NUMBER IS
    total_count NUMBER;
BEGIN
    SELECT COUNT(*) INTO total_count
    FROM employees
    WHERE dept_id = dept;
    RETURN total_count;
END;
/

-- Greeting function
CREATE OR REPLACE FUNCTION get_greeting(name IN VARCHAR2) RETURN VARCHAR2
IS
BEGIN
    RETURN 'Hello, ' || name || '!';
END;
/

-- Main PL/SQL Block
BEGIN
    DECLARE
        emp_count NUMBER;
        greeting VARCHAR2(100);
        fact_val NUMBER;
    BEGIN
        -- a. Call a function directly
        greeting := get_greeting('Oracle Student');
        DBMS_OUTPUT.PUT_LINE('Direct Greeting: ' || greeting);

        -- b. Loop through multiple employee names (to avoid ORA-01422)
        FOR emp_rec IN (
            SELECT emp_name FROM employees WHERE dept_id = 10
        ) LOOP
            greeting := get_greeting(emp_rec.emp_name);
            DBMS_OUTPUT.PUT_LINE('Greeting from loop: ' || greeting);
        END LOOP;

        -- c. Factorial of 5
        fact_val := factorial(5);
        DBMS_OUTPUT.PUT_LINE('Factorial of 5 is: ' || fact_val);

        -- d. Count employees in department 10
        emp_count := count_employees(10);
        DBMS_OUTPUT.PUT_LINE('Total employees in department 10: ' || emp_count);

        -- e. Store factorial result into variable
        DECLARE
            result NUMBER;
        BEGIN
            result := factorial(4);
            DBMS_OUTPUT.PUT_LINE('Factorial of 4 is: ' || result);
        END;
    END;
END;
