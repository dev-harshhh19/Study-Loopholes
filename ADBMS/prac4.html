Program 1: Using GOTO to Exit a Loop Early

Write a PL/SQL program to print numbers from 1 to 10, but use GOTO to exit the loop when the number is 5.

DECLARE
    i NUMBER := 1;
BEGIN
    <>
    LOOP
        DBMS_OUTPUT.PUT_LINE('i = ' || i);
        IF i = 5 THEN
            GOTO exit_point;
        END IF;
        i := i + 1;
    END LOOP my_loop;

    <>
    NULL; -- Just a placeholder to jump to
    DBMS_OUTPUT.PUT_LINE('Exited loop using GOTO at i = ' || i);
END;

Program 3: Combined GOTO and NULL

Write a PL/SQL program to iterate from 1 to 10. If the number is divisible by 4, skip printing using GOTO. USE NULL to do nothing when the number is odd.

DECLARE
    i NUMBER := 1;
BEGIN
    <>
    LOOP
        EXIT WHEN i > 10;

        IF MOD(i, 4) = 0 THEN
            GOTO skip_print;
        ELSIF MOD(i, 2) = 1 THEN
            NULL; -- odd numbers: do nothing
        ELSE
            DBMS_OUTPUT.PUT_LINE('Even number: ' || i);
        END IF;

        <>
        NULL;

        i := i + 1;
    END LOOP main_loop;
END;

Program 6: WHILE Loop with GOTO

Write a PL/SQL block to print numbers 1 to 5, but stop using GOTO when 3 is reached.

DECLARE
    n NUMBER := 1;
BEGIN
    WHILE n <= 5 LOOP
        IF n = 3 THEN
            GOTO end_loop;
        END IF;
        DBMS_OUTPUT.PUT_LINE('Number: ' || n);
        n := n + 1;
    END LOOP;

    <>
    DBMS_OUTPUT.PUT_LINE('Loop exited at number 3');
END;

Program 7: GOTO Inside Nested Loops

Exit both loops using GOTO when condition is met.

DECLARE
    i NUMBER;
    j NUMBER;
BEGIN
    FOR i IN 1..3 LOOP
        FOR j IN 1..3 LOOP
            DBMS_OUTPUT.PUT_LINE('i = ' || i || ', j = ' || j);
            IF i = 2 AND j = 2 THEN
                GOTO exit_all;
            END IF;
        END LOOP;
    END LOOP;

    <>
    DBMS_OUTPUT.PUT_LINE('Exited all loops at i=2, j=2');
END;
/
