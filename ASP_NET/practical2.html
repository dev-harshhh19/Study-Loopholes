Practical2A.aspx.cs
using System;

// Base Class
public class Animal
{
    public string Eat() { return "Eating..."; }
}

// Single Inheritance
public class Dog : Animal
{
    public string Bark() { return "Barking..."; }
}

// Multilevel Inheritance
public class Puppy : Dog
{
    public string Weep() { return "Weeping..."; }
}

// Interface
interface IShape
{
    string Draw();
}

class Circle : IShape
{
    public string Draw() { return "Drawing Circle"; }
}

public partial class Practical2A : System.Web.UI.Page
{
    protected void Page_Load(object sender, EventArgs e) { }

    protected void btnInherit_Click(object sender, EventArgs e)
    {
        Puppy p = new Puppy();
        lblResult.Text = "Inheritance Demo:<br/>" +
            "Puppy.Eat(): " + p.Eat() + "<br/>" +
            "Puppy.Bark(): " + p.Bark() + "<br/>" +
            "Puppy.Weep(): " + p.Weep();
    }

    protected void btnInterface_Click(object sender, EventArgs e)
    {
        IShape shape = new Circle();
        lblResult.Text = "Interface Demo:<br/>" + shape.Draw();
    }
}

using System;

// Delegate Definition
public delegate int MathOperation(int a, int b);

public partial class Practical2B : System.Web.UI.Page
{
    protected void Page_Load(object sender, EventArgs e) { }

    // Method matching delegate signature
    public int Add(int x, int y) { return x + y; }

    protected void btnDelegate_Click(object sender, EventArgs e)
    {
        MathOperation op = new MathOperation(Add);
        int result = op(10, 20);
        lblResult.Text = "Delegate Result: Add(10,20) = " + result;
    }

    protected void btnException_Click(object sender, EventArgs e)
    {
        try
        {
            int divisor = Convert.ToInt32(txtDivisor.Text);
            int result = 100 / divisor;
            lblResult.Text = "100 / " + divisor + " = " + result;
        }
        catch (DivideByZeroException)
        {
            lblResult.Text = "Error: Cannot divide by zero!";
        }
        catch (FormatException)
        {
            lblResult.Text = "Error: Enter a valid number.";
        }
        finally
        {
            lblResult.Text += "<br/>(Finally block executed)";
        }
    }
}
