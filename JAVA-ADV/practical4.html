Practical 4: Applet Logic

Aim

Part A: Write a Java Program to create an applet that receives an integer in one text field, and computes its factorial value and returns it in another text field, when the button named "Compute" is clicked.

Part B: Write a Java Program to create an applet that receives a string and returns either it Uppercase or Lowercase, Reverse of it.

Source Code

FactorialApplet.java
java 
import java.applet.Applet;
import java.awt.Button;
import java.awt.Label;
import java.awt.TextField;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

/* 
 * <applet code="FactorialApplet.class" width="450" height="150">
 * </applet> 
 */

public class FactorialApplet extends Applet implements ActionListener {
    TextField inputField, resultField;
    Button btnCompute;

    public void init() {
        add(new Label("Enter a Number:"));
        
        inputField = new TextField(10);
        add(inputField);
        
        btnCompute = new Button("Compute");
        add(btnCompute);
        btnCompute.addActionListener(this);
        
        add(new Label("Factorial Result:"));
        
        resultField = new TextField(20);
        resultField.setEditable(false); // Make it read-only
        add(resultField);
    }

    public void actionPerformed(ActionEvent e) {
        try {
            int number = Integer.parseInt(inputField.getText());
            long fact = 1;
            
            // Simple Factorial Loop
            for (int i = 1; i <= number; i++) {
                fact = fact * i;
            }
            
            resultField.setText(String.valueOf(fact));
        } catch(NumberFormatException ex) {
            resultField.setText("Error: Not a valid number!");
        }
    }
}
StringOps.java
java 
import java.applet.Applet;
import java.awt.Button;
import java.awt.Label;
import java.awt.TextField;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

/* 
 * <applet code="StringOps.class" width="500" height="200">
 * </applet> 
 */

public class StringOps extends Applet implements ActionListener {
    TextField txtInput, txtOutput;
    Button btnUpper, btnLower, btnReverse;

    public void init() {
        add(new Label("Original String: "));
        txtInput = new TextField(30);
        add(txtInput);
        
        btnUpper = new Button("Uppercase");
        btnLower = new Button("Lowercase");
        btnReverse = new Button("Reverse");
        
        add(btnUpper); 
        add(btnLower); 
        add(btnReverse);
        
        btnUpper.addActionListener(this);
        btnLower.addActionListener(this);
        btnReverse.addActionListener(this);
        
        add(new Label("Final Result: "));
        txtOutput = new TextField(30);
        txtOutput.setEditable(false);
        add(txtOutput);
    }

    public void actionPerformed(ActionEvent e) {
        String str = txtInput.getText();
        Object source = e.getSource();
        
        if (source == btnUpper) {
            txtOutput.setText(str.toUpperCase());
        } else if (source == btnLower) {
            txtOutput.setText(str.toLowerCase());
        } else if (source == btnReverse) {
            // Using StringBuilder to easily reverse the string
            StringBuilder sb = new StringBuilder(str);
            txtOutput.setText(sb.reverse().toString());
        }
    }
}
