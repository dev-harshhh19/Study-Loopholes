Practical 2: Applet Drawings

Aim

Part A: Write a java program to draw Lines, ovals, filled ovals and arcs, filled arcs.

Part B: Write a java program to draw rectangle, filled rectangle and rounded rectangle and filled rounded rectangle with any two colors.

Part C: Write a java program to draw a smiley face.

Source Code

DrawShapes.java
java 
import java.applet.Applet;
import java.awt.Color;
import java.awt.Graphics;

/* 
 * <applet code="DrawShapes.class" width="500" height="400">
 * </applet> 
 */

public class DrawShapes extends Applet {

    public void init() {
        setBackground(Color.white);
    }

    public void paint(Graphics g) {
        g.setColor(Color.black);
        g.drawString("Drawing Lines, Ovals, and Arcs", 150, 20);

        // Drawing Lines
        g.setColor(Color.red);
        g.drawString("draw line:", 20, 60);
        g.drawLine(100, 50, 400, 50);
        g.drawLine(100, 60, 400, 60); // another line parallel

        // Drawing Ovals
        g.setColor(Color.blue);
        g.drawString("draw oval:", 20, 130);
        g.drawOval(100, 100, 100, 50); // width 100, height 50
        g.drawOval(250, 80, 80, 80);   // width 80, height 80 (perfect circle)

        // Drawing Filled Ovals
        g.setColor(Color.magenta);
        g.drawString("fill oval:", 20, 230);
        g.fillOval(100, 200, 100, 50);
        g.fillOval(250, 180, 80, 80);

        // Drawing Arcs
        g.setColor(Color.darkGray);
        g.drawString("draw arc:", 20, 310);
        g.drawArc(100, 280, 80, 80, 0, 180); // semi circle
        g.drawArc(200, 280, 80, 80, 0, 90);  // quarter circle

        // Drawing Filled Arcs
        g.setColor(Color.orange);
        g.drawString("fill arc:", 300, 310);
        g.fillArc(380, 280, 80, 80, 0, 270); // pie shape
    }
}
Rectangles.java
java 
import java.applet.Applet;
import java.awt.Color;
import java.awt.Graphics;

/* 
 * <applet code="Rectangles.class" width="500" height="400">
 * </applet> 
 */

public class Rectangles extends Applet {
    
    public void init() {
        setBackground(Color.white);
    }
    
    public void paint(Graphics g) {
        g.drawString("Drawing Various Rectangles", 150, 20);

        // 1. Standard outline rectangle
        g.setColor(Color.red);
        g.drawString("drawRect", 50, 60);
        g.drawRect(50, 80, 150, 80);

        // 2. Standard filled rectangle
        g.setColor(Color.blue);
        g.drawString("fillRect", 280, 60);
        g.fillRect(280, 80, 150, 80);
        
        // 3. Rounded outline rectangle
        g.setColor(Color.green);
        g.drawString("drawRoundRect", 50, 220);
        // last two parameters determine the arc width & height for rounded corners
        g.drawRoundRect(50, 240, 150, 80, 30, 30);
        
        // 4. Rounded filled rectangle
        g.setColor(Color.magenta);
        g.drawString("fillRoundRect", 280, 220);
        g.fillRoundRect(280, 240, 150, 80, 30, 30);
    }
}
Smiley.java
java 
import java.applet.Applet;
import java.awt.Color;
import java.awt.Graphics;

/* 
 * <applet code="Smiley.class" width="400" height="400">
 * </applet> 
 */

public class Smiley extends Applet {

    public void init() {
        setBackground(Color.white);
    }

    public void paint(Graphics g) {
        // Base Face Circle (Yellow)
        g.setColor(Color.yellow);
        g.fillOval(50, 50, 300, 300);
        
        // Outline of the Face (Black)
        g.setColor(Color.black);
        g.drawOval(50, 50, 300, 300);
        
        // Left Eye
        g.fillOval(120, 130, 40, 40); 
        
        // Right Eye
        g.fillOval(240, 130, 40, 40); 
        
        // Nose (Optional - just to add more detail)
        g.drawLine(200, 180, 200, 220);
        
        // Smile using Arc
        // drawArc(x, y, width, height, startAngle, arcAngle)
        // Note: Negative arcAngle draws clockwise, making a smile.
        g.setColor(Color.red);
        g.fillArc(120, 180, 160, 100, 0, -180);
        
        // Outline for smile to pop more
        g.setColor(Color.black);
        g.drawArc(120, 180, 160, 100, 0, -180);
        
        // Optional text
        g.drawString("Have a Great Day!", 140, 380);
    }
}
