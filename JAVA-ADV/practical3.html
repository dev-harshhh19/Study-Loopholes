Practical 3: Event Handling

Aim

Part A: Write a Java program to create an Applet that displays 4 buttons each represents different colors. If a user click on particular button then that color is set as back ground to applet.

Part B: Write a Java program to create an Applet that displays 2 buttons. If a user clicks on one button then change name of another button and display the button clicked and vice-versa.

Part C: Write a Java program to create an Applet that displays 2 TextFields and labels User Name and Password. Display Text Field entered and display selected text in username & display password.

Part D: Write a Java program that displays 4 buttons and also count Number of button clicks on each button.

ColorButtons.java
java 
import java.applet.Applet;
import java.awt.Button;
import java.awt.Color;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

/* 
 * <applet code="ColorButtons.class" width="400" height="300">
 * </applet> 
 */

public class ColorButtons extends Applet implements ActionListener {
    Button btnRed, btnGreen, btnBlue, btnYellow;

    public void init() {
        btnRed = new Button("Red");
        btnGreen = new Button("Green");
        btnBlue = new Button("Blue");
        btnYellow = new Button("Yellow");

        btnRed.setBackground(Color.red);
        btnGreen.setBackground(Color.green);
        btnBlue.setBackground(Color.blue);
        btnYellow.setBackground(Color.yellow);
        
        btnBlue.setForeground(Color.white);
        btnRed.setForeground(Color.white);

        add(btnRed);
        add(btnGreen);
        add(btnBlue);
        add(btnYellow);
        
        // Register the buttons to the Action Listener
        btnRed.addActionListener(this);
        btnGreen.addActionListener(this);
        btnBlue.addActionListener(this);
        btnYellow.addActionListener(this);
    }

    // Handles the button click events
    public void actionPerformed(ActionEvent e) {
        String command = e.getActionCommand();
        
        if (command.equals("Red")) {
            setBackground(Color.red);
        } else if (command.equals("Green")) {
            setBackground(Color.green);
        } else if (command.equals("Blue")) {
            setBackground(Color.blue);
        } else if (command.equals("Yellow")) {
            setBackground(Color.yellow);
        }
        
        // Force the applet to redraw its UI
        repaint(); 
    }
}
ToggleButtons.java
java 
import java.applet.Applet;
import java.awt.Button;
import java.awt.Font;
import java.awt.Graphics;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

/* 
 * <applet code="ToggleButtons.class" width="400" height="300">
 * </applet> 
 */

public class ToggleButtons extends Applet implements ActionListener {
    Button b1, b2;
    String statusMessage = "Click a button above.";

    public void init() {
        b1 = new Button("Button A");
        b2 = new Button("Button B");
        
        add(b1);
        add(b2);
        
        b1.addActionListener(this);
        b2.addActionListener(this);
    }

    public void actionPerformed(ActionEvent e) {
        // Did user click Button 1?
        if (e.getSource() == b1) {
            b2.setLabel("I was changed by A!");
            statusMessage = "You just clicked on the First Button.";
        } 
        // Or did user click Button 2?
        else if(e.getSource() == b2) {
            b1.setLabel("I was changed by B!");
            statusMessage = "You just clicked on the Second Button.";
        }
        repaint();
    }

    public void paint(Graphics g) {
        g.setFont(new Font("Arial", Font.BOLD, 16));
        g.drawString(statusMessage, 50, 150);
    }
}
LoginApplet.java
java 
import java.applet.Applet;
import java.awt.Button;
import java.awt.Label;
import java.awt.TextField;
import java.awt.Graphics;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

/* 
 * <applet code="LoginApplet.class" width="400" height="250">
 * </applet> 
 */

public class LoginApplet extends Applet implements ActionListener {
    TextField txtUsername, txtPassword;
    Button btnSubmit;
    String displayMsg = "";

    public void init() {
        // Label & TextField for Username
        add(new Label("User Name: "));
        txtUsername = new TextField(20);
        add(txtUsername);
        
        // Label & TextField for Password
        add(new Label("Password: "));
        txtPassword = new TextField(20);
        txtPassword.setEchoChar('*'); // Hides password typing
        add(txtPassword);
        
        btnSubmit = new Button("Login / Submit");
        add(btnSubmit);
        
        btnSubmit.addActionListener(this);
    }

    public void actionPerformed(ActionEvent e) {
        String uName = txtUsername.getText();
        String uPass = txtPassword.getText();
        
        // This is only for demonstrating UI update
        displayMsg = "Output - User: " + uName + " | Pass: " + uPass; 
        repaint();
    }

    public void paint(Graphics g) {
        g.drawString(displayMsg, 20, 150);
    }
}
ButtonCounts.java
java 
import java.applet.Applet;
import java.awt.Button;
import java.awt.Graphics;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

/* 
 * <applet code="ButtonCounts.class" width="400" height="250">
 * </applet> 
 */

public class ButtonCounts extends Applet implements ActionListener {
    Button b1, b2, b3, b4;
    int c1 = 0, c2 = 0, c3 = 0, c4 = 0; // The four counters

    public void init() {
        b1 = new Button("Option 1"); 
        b2 = new Button("Option 2");
        b3 = new Button("Option 3"); 
        b4 = new Button("Option 4");
        
        add(b1); 
        add(b2); 
        add(b3); 
        add(b4);
        
        b1.addActionListener(this); 
        b2.addActionListener(this);
        b3.addActionListener(this); 
        b4.addActionListener(this);
    }

    public void actionPerformed(ActionEvent e) {
        Object source = e.getSource();
        
        if (source == b1) {
            c1++;
        } else if(source == b2) {
            c2++;
        } else if(source == b3) {
            c3++;
        } else if(source == b4) {
            c4++;
        }
        
        repaint();
    }

    public void paint(Graphics g) {
        g.drawString("Option 1 Clicks: " + c1, 50, 100);
        g.drawString("Option 2 Clicks: " + c2, 50, 130);
        g.drawString("Option 3 Clicks: " + c3, 50, 160);
        g.drawString("Option 4 Clicks: " + c4, 50, 190);
    }
}
