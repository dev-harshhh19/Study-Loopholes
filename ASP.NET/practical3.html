Practical3A.aspx.cs
using System;

public partial class Practical3A : System.Web.UI.Page
{
    protected void Page_Load(object sender, EventArgs e) { }

    protected void btnSubmit_Click(object sender, EventArgs e)
    {
        string msg = "Hello " + txtName.Text;
        
        if (chkTerms.Checked)
            msg += " (Terms Accepted)";
        
        if (rbMale.Checked)
            msg += " [Male]";
        else if (rbFemale.Checked)
            msg += " [Female]";
        
        lblResult.Text = msg;
    }
}

Practical3B.aspx.cs
using System;
using System.Web.UI.WebControls;

public partial class Practical3B : System.Web.UI.Page
{
    protected void Page_Load(object sender, EventArgs e) { }

    protected void ddlLanguage_Changed(object sender, EventArgs e)
    {
        lblResult.Text = "Language: " + ddlLanguage.SelectedValue;
    }

    protected void btnShow_Click(object sender, EventArgs e)
    {
        string selected = "Frameworks: ";
        foreach (ListItem item in lstFramework.Items)
        {
            if (item.Selected)
                selected += item.Text + ", ";
        }
        lblResult.Text = selected.TrimEnd(',', ' ');
    }
}
